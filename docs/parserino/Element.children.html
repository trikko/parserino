    Function Element.children

    All the children contained in this element. (non-deep search)

    Example

    Document d = "<p><b>test</b>test2</p>";

import std.array;
Element[] c = d.body.firstChild.children(true).array;
assert(c.length == 2);
assert(c[0] == "<b>test</b>");
assert(c[1] == "test2");
